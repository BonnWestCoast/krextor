    <!-- the mcd:prototype element -->

    <param name="prototype" />

    <om:_>

        <for-each select="'cdbase', 'cd', 'name'">

        <variable name="attribute" select="$symbol/@*[local-name() eq current()]" />

        <if test="$attribute">

            <attribute name="{current()}" select="$attribute" />

        </if>

    </function>
